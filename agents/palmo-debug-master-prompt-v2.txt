# Race condition - 20 concurrent POSTs
for i in $(seq 1 20); do
  curl -s -o /tmp/rc-$i.json -X POST http://localhost:5000/api/submissions \
    -H "Content-Type: application/json" \
    -d "{\"weight\":$i,\"contactEmail\":\"rc$i@t.de\",\"pickupAddress\":\"Berlin\",\"pickupCoords\":{\"lat\":52.52,\"lng\":13.41},\"deliveryAddress\":\"Warsaw\",\"deliveryCoords\":{\"lat\":52.23,\"lng\":21.01},\"serviceType\":\"standard\",\"additionalServices\":{},\"estimatedDistance\":573000,\"estimatedDuration\":19800,\"estimatedPrice\":100}" &
done
wait
python3 -c "
import json,glob
ids=[]
for f in glob.glob('/tmp/rc-*.json'):
    try:
        d=json.load(open(f))
        id_=(d.get('data') or d).get('id')
        if id_: ids.append(id_)
    except: pass
dupes=len(ids)-len(set(ids))
print(f'{len(ids)}/20 OK | Duplicates: {dupes}')
print('PASS' if dupes==0 else 'FAIL: Race condition!')
"

## FAZA 5: SECURITY

# Security headers (wymaga helmet.js)
HEADERS=$(curl -sI http://localhost:5000/health)
for H in "X-Content-Type-Options" "X-Frame-Options" "Content-Security-Policy"; do
  echo "$HEADERS" | grep -qi "$H" && echo "OK: $H" || echo "MISSING: $H -> npm install helmet"
done
echo "$HEADERS" | grep -qi "x-powered-by: Express" \
  && echo "BUG: Server fingerprint exposed -> app.disable('x-powered-by')" \
  || echo "OK: fingerprint hidden"

# Rate limiting (wyslij 100 req - sprawdz czy jest 429)
BLOCKED=0
for i in $(seq 1 100); do
  [ "$(curl -s -o /dev/null -w "%{http_code}" http://localhost:5000/api/submissions)" = "429" ] \
    && BLOCKED=$((BLOCKED+1))
done
[ $BLOCKED -gt 0 ] && echo "OK: Rate limiting active ($BLOCKED blocked)" || echo "WARNING: No rate limiting -> npm install express-rate-limit"

# WordPress auth check
WP_STATUS=$(curl -s -o /dev/null -w "%{http_code}" http://localhost:8088/wp-json/palmo/v1/submissions)
[ "$WP_STATUS" = "401" ] || [ "$WP_STATUS" = "403" ] \
  && echo "OK: WP auth required" \
  || echo "WARNING: WP endpoint public ($WP_STATUS) - add permission_callback"
